cmake_minimum_required(VERSION 3.5)

project(search_engine_3 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(JSON_BuildTests OFF CACHE INTERNAL "")
set(gtest_disable_pthreads on)

add_subdirectory(nlohmann_json)
add_executable(
    search_engine_3
    main.cpp
    ${CMAKE_SOURCE_DIR}/config/answers.json
    ${CMAKE_SOURCE_DIR}/config/config.json
    ${CMAKE_SOURCE_DIR}/configrequests.json
    ${CMAKE_SOURCE_DIR}/include/SearchServer.h
    ${CMAKE_SOURCE_DIR}/include/ConvertJSON.h
    ${CMAKE_SOURCE_DIR}/include/InvertedIndex.h
    README.md

)
target_link_libraries(search_engine_3 PRIVATE nlohmann_json::nlohmann_json)
